class Solution {
    public int[] solution(int brown, int yellow) {
        int sum = brown + yellow;
        int[] answer = new int[2];
        
        for (int i = 3; i < sum; i++) { //가로와 세로는 3이상이어야함
            
            if (sum % i == 0) {
                int ga = sum/i; //가로
                int se = sum/ga; //세로
                
                int a = ga - 2;
                int b = se - 2;
                
                if(a*b == yellow && ga>=se){ //가로 >= 세로
                    answer[0] = ga;
                    answer[1] = se;
                }
            }
        }
        return answer;
    }
}
